116    @Override
117    public void onEnable() {
118        instance = this;
119
120        for (Plugin plugin : Bukkit.getPluginManager().getPlugins()) {
121            if (plugin instanceof CalibreHook)
122                hooks.add((CalibreHook) plugin);
123        }
124
125        hooks.forEach(hook -> hook.onEnable(this));
126
127        configOptions = ConfigurationOptions.defaults()
128                .serializers(builder -> {
129                    hooks.forEach(hook -> hook.registerSerializers(builder));
130                    statMapSerializer = new StatMapSerializer(Utils.serializer(StatMap.class, builder), key -> setting("stat_config", key));
131                    builder
132                            .register(Color.class, ColorSerializer.INSTANCE)
133                            .register(ItemStack.class, ItemStackSerializer.INSTANCE)
134                            .register(Location.class, LocationSerializer.INSTANCE)
135                            .register(NamespacedKey.class, NamespacedKeySerializer.INSTANCE)
136                            .register(DoubleDescriptor.class, DoubleDescriptorSerializer.INSTANCE)
137                            .register(Vector2DDescriptor.class, Vector2DDescriptorSerializer.INSTANCE)
138                            .register(Vector.class, VectorSerializer.INSTANCE)
139                            .register(GUIVector.class, GUIVectorSerializer.INSTANCE)
140                            .register(Vector3D.class, Vector3DSerializer.INSTANCE)
141                            .register(Vector3I.class, Vector3ISerializer.INSTANCE)
142                            .register(Vector2D.class, Vector2DSerializer.INSTANCE)
143                            .register(Vector2I.class, Vector2ISerializer.INSTANCE)
144                            .register(ParticleData.class, ParticleDataSerializer.INSTANCE)
145                            .register(StatMap.class, statMapSerializer)
146                            .register(StatCollection.class, StatCollection.Serializer.INSTANCE)
147
148                            .register(new TypeToken<ComponentContainerSystem<?>>(){}, new ComponentContainerSystem.Serializer(builder.build().get(new TypeToken<ComponentContainerSystem<?>>(){})))
149                            .register(new TypeToken<Quantifier<ComponentTree>>(){}, new QuantifierSerializer<>())
150                            .register(SightRef.class, SightRef.Serializer.INSTANCE)
151                            .register(FireModeRef.class, FireModeRef.Serializer.INSTANCE)
152                            .register(PaperComponent.class, new PaperComponent.Serializer(Utils.serializer(PaperComponent.class, builder)))
153                            .register(ComponentTree.class, new ComponentTree.AbstractSerializer() {
154                                @Override protected <T extends CalibreIdentifiable> T byId(String id, Class<T> type) { return registry.get(id, type); }
155                            })
156                            .registerAnnotatedObjects(mapperFactory);
157                });
158
159        commandManager = new PaperCommandManager(this);
160        commandManager.enableUnstableAPI("help");
161        commandManager.getCommandCompletions().registerCompletion("registry", ctx -> {
162            String sType = ctx.getConfig("type");
163            Class<?> type = null;
164            if (sType != null) {
165                try {
166                    type = Class.forName(sType);
167                } catch (ClassNotFoundException | ClassCastException ignore) {}
168            }
169
170            final Class<?> fType = type;
171            return registry.getRegistry().entrySet().stream()
172                    .filter(entry -> fType == null || fType.isInstance(entry.getValue().get()))
173                    .map(Map.Entry::getKey)
174                    .collect(Collectors.toList());
175        });
176        commandManager.getCommandContexts().registerContext(CalibreIdentifiable.class, ctx -> {
177            String sType = ctx.getFlagValue("type", (String) null);
178            Class<?> type = null;
179            if (sType != null) {
180                try {
181                    type = Class.forName(sType);
182                } catch (ClassNotFoundException | ClassCastException ignore) {}
183            }
184
185            CommandSender sender = ctx.getSender();
186            String locale = locale(sender);
187            String id = ctx.popFirstArg();
188            Ref<? extends CalibreIdentifiable> result = registry.getRef(id);
189            if (result == null) {
190                sendMessage(sender, gen(locale, "command.error.no_object",
191                        "id", id));
192                throw new InvalidCommandArgument(false);
193            }
194            CalibreIdentifiable obj = result.get();
195            if (type != null && !type.isInstance(result.get())) {
196                sendMessage(sender, gen(locale, "command.error.not_type",
197                        "id", id,
198                        "found", obj.getClass().getSimpleName(),
199                        "expected", type.getSimpleName()));
200                throw new InvalidCommandArgument(false);
201            }
202            return obj;
203        });
204        commandManager.getCommandContexts().registerContext(ComponentTree.class, ctx -> {
205            CommandSender sender = ctx.getSender();
206            String locale = locale(sender);
207            String json = ctx.popFirstArg();
208
209            ComponentTree tree;
210            try {
211                tree = ComponentTree.deserialize(json, configOptions);
212            } catch (ConfigurateException e) {
213                sendMessage(sender, gen(locale, "command.error.parse_tree",
214                        "error", TextUtils.combineMessages(e)));
215                throw new InvalidCommandArgument(false);
216            }
217
218            if (tree == null) {
219                sendMessage(sender, gen(locale, "command.error.parse_tree",
220                        "error", "null"));
221                throw new InvalidCommandArgument(false);
222            }
223
224            return tree;
225        });
226        commandManager.registerCommand(new CalibreCommand(this));
227
228        guiManager = new GUIManager(this);
229
230        audiences = BukkitAudiences.create(this);
231
232        protocol = ProtocolLibrary.getProtocolManager();
233
234        font = new MinecraftFont();
235
236        Bukkit.getPluginManager().registerEvents(new CalibreListener(this), this);
237
238        hooks.forEach(hook -> hook.onRegistryLoad(registry));
239        // Defaults
240        registry.onLoad(r -> {
241            r.register(new PaperGunSystem(this));
242            r.register(new PaperStatDisplaySystem(this));
243            r.register(new PaperSlotDisplaySystem(this));
244            r.register(new PaperSightSystem(this));
245            r.register(new PaperFireModeSystem(this));
246            r.register(new PaperComponentContainerSystem(this));
247            r.register(new PaperCapacityComponentContainerSystem(this));
248            r.register(new PaperNameFromChildSystem(this));
249        });
250
251        StatDisplaySystem.renderer(DoubleDescriptor.class, (inst, locale, key) -> {
252            DoubleDescriptor desc = (DoubleDescriptor) inst.raw();
253            DoubleDescriptorStat stat = (DoubleDescriptorStat) inst.stat();
254            Component bar = null;
255            if (desc.operation() == NumericalDescriptor.Operation.SET && stat.min() != null && stat.max() != null) {
256                double min = stat.min();
257                double max = stat.max();
258                double value = desc.value();
259                bar = bar(
260                        locale, "system.stat_display.bar." + key,
261                        (value - min) / (max - min), 0,
262                        setting("system", StatDisplaySystem.ID, "bar_lengths").node(0).getInt()
263                );
264            }
265            Component value = Component.text(stat.valueToString(desc));
266            return bar == null ? value : bar.append(value);
267        });
268        StatDisplaySystem.renderer(Vector2DDescriptor.class, (inst, locale, key) -> {
269            Vector2DDescriptor desc = (Vector2DDescriptor) inst.raw();
270            Vector2DDescriptorStat stat = (Vector2DDescriptorStat) inst.stat();
271            Component bar = null;
272            if (desc.operation() == NumericalDescriptor.Operation.SET && stat.min() != null && stat.max() != null) {
273                double min = stat.min();
274                double max = stat.max();
275                double x = desc.value().x();
276                double y = desc.value().y();
277
278                int barLength = setting("system", StatDisplaySystem.ID, "bar_lengths").node(1).getInt();
279                bar = Component.text()
280                        .append(bar(locale, "system.stat_display.bar." + key, (x - min) / (max - min), 0, barLength))
281                        .append(bar(locale, "system.stat_display.bar." + key, (y - min) / (max - min), 0, barLength))
282                        .build();
283            }
284            Component value = Component.text(stat.valueToString(desc));
285            return bar == null ? value : bar.append(value);
286        });
287
288        saveDefaults();
289        logAll(load());
290
291        hooks.forEach(CalibreHook::postEnable);
292
293        schedulerLoop = new SchedulerLoop(this);
294        schedulerLoop.nextTick(this);
295        schedulerLoop.start();
296    }

333                .combine(loadLocales())
334                .combine(loadRegistry());

337    private LoggingResult loadSettings() {
338        try {
339            settings = HoconConfigurationLoader.builder()
340                    .file(BukkitUtils.getRelative(this, SETTINGS_FILE))
341                    .defaultOptions(configOptions)
342                    .build()
343                    .load();
344        } catch (ConfigurateException e) {
345            settings = BasicConfigurationNode.root();
346            return new LoggingResult()
347                    .addFailure(LogLevel.ERROR, "Could not load settings from " + SETTINGS_FILE, e);
348        }
349
350        logger.setLevel(LogLevel.valueOfDefault(setting("log_level").getString("VERBOSE")));
351
352        settings.node("font_map").childrenMap().forEach((key, node) ->
353                font.setChar(key.toString().charAt(0), new MapFont.CharacterSprite(node.getInt(), 0, new boolean[0])));
354
355        return new LoggingResult()
356                .addSuccess(LogLevel.INFO, "Loaded settings from " + SETTINGS_FILE);
357    }
358
359    private LoggingResult loadLocales() {
360        LoggingResult result = new LoggingResult();
361        localeManager.unregisterAll();
362
363        LocaleLoader.hocon(BukkitUtils.getRelative(this, "lang"), localeManager).forEach(entry -> {
364            if (entry.isSuccessful())
365                result.addSuccess(LogLevel.INFO, "Loaded " + ((Translation) entry.getResult()).getLocale() + " from " + entry.getKey());
366            else {
367                Exception e = ((Exception) entry.getResult());
368                result.addFailure(LogLevel.WARN, "Could not load locales from " + entry.getKey(), e);
369            }
370        });
371        return result;
372    }
373
374    private LoggingResult loadRegistry() {
375        LoggingResult result = new LoggingResult();
376        registry.unregisterAll();
377
378        new ConfigurateRegistryLoader.Hocon<CalibreIdentifiable>(getDataFolder())
379                .options(configOptions)
380                .with("component", PaperComponent.class)
381                //.with("blueprint", Blueprint.class)
382                .load(registry).forEach(entry -> {
383            if (entry.isSuccessful()) {
384                Identifiable id = ((Ref<?>) entry.getResult()).get();
385                result.addSuccess(LogLevel.VERBOSE, "Loaded " + id.getClass().getSimpleName() + " " + id.id());
386            } else {
387                Exception e = ((Exception) entry.getResult());
388                result.addFailure(LogLevel.WARN, "Could not load registry from " + entry.getKey(), e);
389            }
390        });
391
392        registry.link().forEach(entry -> {
393            Identifiable id = entry.getKey().get();
394            if (entry.isSuccessful()) {
395                Identifiable dep = ((Ref<?>) entry.getResult()).get();
396                result.addSuccess(LogLevel.VERBOSE, "Linked " + id.getClass().getSimpleName() + " " + id.id() +
397                        " with " + dep.getClass().getSimpleName() + " " + dep.id());
398            } else {
399                Exception e = ((Exception) entry.getResult());
400                result.addFailure(LogLevel.WARN, "Could not link " + id.getClass().getSimpleName(), e);
401            }
402        });
403
404        registry.resolve().forEach(entry -> {
405            Identifiable id = entry.getKey().get();
406            if (entry.isSuccessful()) {
407                result.addSuccess(LogLevel.VERBOSE, "Resolved " + id.getClass().getSimpleName() + " " + id.id());
408            } else {
409                Exception e = ((Exception) entry.getResult());
410                result.addFailure(LogLevel.WARN, "Could not resolve " + id.getClass().getSimpleName() + " " + id.id(), e);
411            }
412        });
413
414        registry.removeUnlinked();
415
416        registry.getRegistry().forEach((id, ref) ->
417                result.addSuccess(LogLevel.INFO, "Registered " + ref.get().getClass().getSimpleName() + " " + id));
418
419        return result;
420    }

445    private String repeat(String text, double amount) {
446        return amount < 1 ? "" : text.repeat((int) amount);
447    }

449    public Component bar(String locale, String key, double fullPercent, double partPercent, int width) {
450        String full = setting("symbol", "full_bar").getString("=");
451        String part = setting("symbol", "part_bar").getString("~");
452        String empty = setting("symbol", "empty_bar").getString("-");
453
454        int fullWidth = (int) (Utils.clamp01(fullPercent) * width);
455        int partWidth = (int) ((Utils.clamp01(partPercent + fullPercent) * width) - fullWidth);
456        int emptyWidth = Math.max(0, width - partWidth - fullWidth);
457
458        return gen(locale, key,
459                "full", full.repeat(fullWidth),
460                "part", part.repeat(partWidth),
461                "empty", empty.repeat(emptyWidth));
462    }

464    public PaperComponent getComponent(ItemStack item) throws ConfigurateException {
465        if (item == null || !item.hasItemMeta())
466            return null;
467        PersistentDataContainer container = item.getItemMeta().getPersistentDataContainer();
468        String json = container.get(key("tree"), PersistentDataType.STRING);
469        if (json == null)
470            return null;
471        return ComponentTree.deserialize(json, configOptions).root();
472    }
473
474    public PaperComponent getComponentOrNull(ItemStack item) {
475        if (item == null)
476            return null;
477        try {
478            return getComponent(item);
479        } catch (ConfigurateException e) {
480            return null;
481        }
482    }

493    @Override
494    public void tick(TickContext tickContext) {
495        Bukkit.getOnlinePlayers().forEach(player -> tickPlayer(player, tickContext));
496        tickContext.reschedule(this);
497    }
498
499    public void tickPlayer(Player player, TickContext tickContext) {
500        EntityEquipment equipment = player.getEquipment();
501        for (EquipmentSlot slot : EquipmentSlot.values()) {
502            PaperComponent component = getComponentOrNull(equipment.getItem(slot));
503            if (component != null)
504                component.tree().call(BukkitItemEvents.BukkitEquipped.of(component, player, slot, tickContext));
505        }
506
507        GUIView view = guiManager.getView(player);
508        if (view != null && view.getGUI() instanceof SlotViewGUI) {
509            SlotViewGUI gui = (SlotViewGUI) view.getGUI();
510            if (!gui.check())
511                view.getView().close();
512        }
513    }
