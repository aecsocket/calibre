001package me.aecsocket.calibre.component;
002
003import io.leangen.geantyref.TypeToken;
004import me.aecsocket.calibre.CalibrePlugin;
005import me.aecsocket.calibre.util.ItemCreationException;
006import me.aecsocket.calibre.util.ItemDescriptor;
007import me.aecsocket.calibre.wrapper.BukkitItem;
008import me.aecsocket.unifiedframework.stat.Stat;
009import me.aecsocket.unifiedframework.stat.impl.descriptor.DoubleDescriptorStat;
010import me.aecsocket.unifiedframework.util.BukkitUtils;
011import me.aecsocket.unifiedframework.util.MapInit;
012import org.bukkit.inventory.ItemFlag;
013import org.bukkit.inventory.ItemStack;
014import org.checkerframework.checker.nullness.qual.Nullable;
015import org.spongepowered.configurate.ConfigurationNode;
016import org.spongepowered.configurate.objectmapping.ConfigSerializable;
017import org.spongepowered.configurate.serialize.SerializationException;
018import org.spongepowered.configurate.serialize.TypeSerializer;
019
020import java.lang.reflect.Type;
021import java.util.LinkedHashMap;
022import java.util.Map;
023
024@ConfigSerializable
025public class PaperComponent extends CalibreComponent<BukkitItem> {
026    public static class Serializer implements TypeSerializer<PaperComponent> {
027        private final TypeSerializer<PaperComponent> delegate;
028
029        public Serializer(TypeSerializer<PaperComponent> delegate) {
030            this.delegate = delegate;
031        }
032
033        public TypeSerializer<PaperComponent> delegate() { return delegate; }
034
035        @Override
036        public void serialize(Type type, @Nullable PaperComponent obj, ConfigurationNode node) throws SerializationException {
037            delegate.serialize(type, obj, node);
038        }
039
040        @Override
041        public PaperComponent deserialize(Type type, ConfigurationNode node) throws SerializationException {
042            PaperComponent obj = delegate.deserialize(type, node);
043
044            obj.slots.clear();
045            Map<String, PaperSlot> slots = node.node("slots").get(new TypeToken<>(){});
046            if (slots != null)
047                obj.slots.putAll(slots);
048
049            return obj;
050        }
051    }
052
053    public static final Map<String, Stat<?>> DEFAULT_STATS = MapInit.of(new LinkedHashMap<String, Stat<?>>())
054            .init("item", new ItemDescriptor.Stat())
055            .init("fov_multiplier", new DoubleDescriptorStat(0.1d).hide())
056            .get();
057
058    protected transient final CalibrePlugin plugin;
059
060    public PaperComponent(CalibrePlugin plugin, String id) {
061        super(id);
062        this.plugin = plugin;
063    }
064
065    public PaperComponent() { this(CalibrePlugin.getInstance(), null); }
066
067    public PaperComponent(CalibreComponent<BukkitItem> o, CalibrePlugin plugin) throws SerializationException {
068        super(o);
069        this.plugin = plugin;
070    }
071
072    public PaperComponent(PaperComponent o) throws SerializationException {
073        this(o, o.plugin);
074    }
075
076    @Override public net.kyori.adventure.text.Component localize(String locale, String key, Object... args) { return plugin.gen(locale, key, args); }
077    @Override public Map<String, Stat<?>> defaultStats() { return DEFAULT_STATS; }
078    @Override
079    protected void prepareStatDeserialization(Map<String, Stat<?>> originals) {
080        plugin.getStatMapSerializer().originals(originals);
081    }
082
083    @Override
084    public CalibreComponent<BukkitItem> getComponent(BukkitItem item) {
085        if (item == null)
086            return null;
087        return plugin.getComponentOrNull(item.item());
088    }
089
090    @Override
091    protected BukkitItem createInitial(int amount) throws ItemCreationException {
092        ItemDescriptor descriptor = tree.stat("item");
093        if (descriptor == null)
094            throw new ItemCreationException("No item descriptor");
095        try {
096            return BukkitItem.of(BukkitUtils.modMeta(descriptor.create(amount), meta ->
097                    meta.addItemFlags(ItemFlag.HIDE_ATTRIBUTES, ItemFlag.HIDE_UNBREAKABLE)));
098        } catch (IllegalArgumentException e) {
099            throw new ItemCreationException(e);
100        }
101    }
102
103    public ItemStack create(String locale, ItemStack existing) {
104        BukkitItem result = create(locale, existing.getAmount());
105        return result == null ? null : result.item();
106    }
107
108    @Override public PaperComponent copy() throws SerializationException { return new PaperComponent(this); }
109}
