001package me.aecsocket.calibre.component;
002
003import me.aecsocket.calibre.CalibrePlugin;
004import me.aecsocket.calibre.util.ItemDescriptor;
005import me.aecsocket.calibre.wrapper.BukkitItem;
006import me.aecsocket.unifiedframework.gui.GUIVector;
007import me.aecsocket.unifiedframework.util.BukkitUtils;
008import net.kyori.adventure.text.serializer.legacy.LegacyComponentSerializer;
009import org.bukkit.inventory.ItemStack;
010import org.spongepowered.configurate.objectmapping.ConfigSerializable;
011import org.spongepowered.configurate.serialize.SerializationException;
012
013@ConfigSerializable
014public class PaperSlot extends CalibreSlot {
015    private transient final CalibrePlugin plugin;
016    private GUIVector offset;
017
018    public PaperSlot(CalibrePlugin plugin, boolean required, boolean fieldModifiable, GUIVector offset) {
019        super(required, fieldModifiable);
020        this.plugin = plugin;
021        this.offset = offset;
022    }
023
024    public PaperSlot(CalibrePlugin plugin, boolean required, boolean fieldModifiable) {
025        super(required, fieldModifiable);
026        this.plugin = plugin;
027    }
028
029    public PaperSlot(CalibrePlugin plugin, boolean required, GUIVector offset) {
030        super(required);
031        this.plugin = plugin;
032        this.offset = offset;
033    }
034
035    public PaperSlot(CalibrePlugin plugin, GUIVector offset) {
036        this.plugin = plugin;
037        this.offset = offset;
038    }
039
040    public PaperSlot(CalibrePlugin plugin) {
041        this.plugin = plugin;
042    }
043
044    public PaperSlot() {
045        plugin = CalibrePlugin.getInstance();
046    }
047
048    public PaperSlot(CalibreSlot o, CalibrePlugin plugin, GUIVector offset) throws SerializationException {
049        super(o);
050        this.plugin = plugin;
051        this.offset = offset;
052    }
053
054    public GUIVector offset() { return offset; }
055    public void offset(GUIVector offset) { this.offset = offset; }
056
057    public ItemStack createViewItem(String locale, String slotKey, CalibreComponent<BukkitItem> cursor) {
058        ItemStack item;
059        try {
060            item = plugin.setting("slot_view", "icon",
061                    cursor == null
062                            ? (required ? "required" : "normal")
063                            : (isCompatible(cursor) ? "compatible" : "incompatible")
064            ).get(ItemDescriptor.class).create();
065        } catch (SerializationException e) {
066            return null;
067        }
068
069        return BukkitUtils.modMeta(item, meta -> {
070            // TODO
071            meta.setDisplayName(LegacyComponentSerializer.legacySection().serialize(plugin.gen(locale, "slot_view.slot",
072                    "key", plugin.gen(locale, "slot." + slotKey))));
073        });
074    }
075
076    @Override
077    public CalibreSlot copy() throws SerializationException {
078        return new PaperSlot(super.copy(), plugin, offset);
079    }
080}
