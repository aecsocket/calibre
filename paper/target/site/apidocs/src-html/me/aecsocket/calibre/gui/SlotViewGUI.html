001package me.aecsocket.calibre.gui;
002
003import me.aecsocket.calibre.CalibrePlugin;
004import me.aecsocket.calibre.component.CalibreComponent;
005import me.aecsocket.calibre.component.PaperComponent;
006import me.aecsocket.calibre.component.PaperSlot;
007import me.aecsocket.calibre.world.ItemSlot;
008import me.aecsocket.calibre.wrapper.BukkitItem;
009import me.aecsocket.unifiedframework.gui.*;
010import org.bukkit.Bukkit;
011import org.bukkit.entity.Player;
012import org.bukkit.event.inventory.InventoryClickEvent;
013import org.spongepowered.configurate.ConfigurateException;
014import org.spongepowered.configurate.serialize.SerializationException;
015
016import java.util.HashMap;
017import java.util.Map;
018
019public class SlotViewGUI extends GUI {
020    public static final InventorySize SIZE = new InventorySize(6);
021    public static final GUIVector CENTER = new GUIVector(4, 3);
022
023    private final CalibrePlugin plugin;
024    private PaperComponent component;
025    private boolean modification;
026    private boolean limited;
027    private ItemSlot<BukkitItem> slot;
028
029    public SlotViewGUI(CalibrePlugin plugin, PaperComponent component, boolean modification, boolean limited, ItemSlot<BukkitItem> slot) {
030        this.plugin = plugin;
031        this.component = component;
032        this.modification = modification;
033        this.limited = limited;
034        this.slot = slot;
035    }
036
037    public CalibrePlugin plugin() { return plugin; }
038
039    public PaperComponent component() { return component; }
040    public void component(PaperComponent component) { this.component = component; }
041
042    public boolean modification() { return modification; }
043    public void modification(boolean modification) { this.modification = modification; }
044
045    public boolean limited() { return limited; }
046    public void limited(boolean limited) { this.limited = limited; }
047
048    public ItemSlot<BukkitItem> slot() { return slot; }
049    public void slot(ItemSlot<BukkitItem> slot) { this.slot = slot; }
050
051    @Override public GUIManager getGUIManager() { return plugin().getGUIManager(); }
052    @Override public InventorySize getSize(Player player) { return SIZE; }
053    @Override public net.kyori.adventure.text.Component getTitle(Player player) {
054        return plugin.gen(player.getLocale(), "slot_view.title",
055            "name", component.name(player.getLocale()));
056    }
057
058    @Override
059    public Map<Integer, GUIItem> getItems(Player player) {
060        Map<Integer, GUIItem> result = new HashMap<>();
061        GUIVector vec;
062        try {
063            vec = plugin.setting("slot_view", "center").get(GUIVector.class, CENTER);
064        } catch (SerializationException e) {
065            vec = CENTER;
066        }
067        result.put(
068                vec.slot(),
069                new SlotViewItem(plugin, component, modification, limited)
070        );
071        for (Map.Entry<String, PaperSlot> entry : component.<PaperSlot>slots().entrySet())
072            addItems(result, entry.getKey(), entry.getValue(), vec);
073        return result;
074    }
075
076    private void addItems(Map<Integer, GUIItem> map, String slotKey, PaperSlot slot, GUIVector offset) {
077        if (slot.offset() != null)
078            offset = new GUIVector(offset.add(slot.offset()));
079        map.put(offset.slot(), new SlotViewItem(plugin, slot, slotKey, modification, limited));
080        if (slot.get() == null)
081            return;
082
083        for (Map.Entry<String, PaperSlot> entry : slot.get().<PaperSlot>slots().entrySet())
084            addItems(map, entry.getKey(), entry.getValue(), offset);
085    }
086
087    public boolean check() {
088        if (slot == null)
089            return true;
090        try {
091            if (slot.get() == null)
092                return false;
093            CalibreComponent<?> realComponent = plugin.getComponent(slot.get().item());
094            return component.equals(realComponent);
095        } catch (ConfigurateException e) {
096            return false;
097        }
098    }
099
100    @Override
101    public void onClick(GUIView view, InventoryClickEvent event) {
102        super.onClick(view, event);
103        Bukkit.getScheduler().scheduleSyncDelayedTask(plugin, view::updateItems, 1);
104    }
105
106    public void notifyUpdate(GUIView view) {
107        component.buildTree();
108        if (slot != null)
109            slot.set(component.create(view.getPlayer().getLocale(), slot.get().amount()));
110        view.reopen();
111    }
112}
