001package me.aecsocket.calibre.gui;
002
003import me.aecsocket.calibre.CalibrePlugin;
004import me.aecsocket.calibre.component.CalibreComponent;
005import me.aecsocket.calibre.component.ComponentTree;
006import me.aecsocket.calibre.component.PaperSlot;
007import me.aecsocket.calibre.wrapper.BukkitItem;
008import me.aecsocket.unifiedframework.gui.GUIItem;
009import me.aecsocket.unifiedframework.gui.GUIView;
010import me.aecsocket.unifiedframework.util.BukkitUtils;
011import net.kyori.adventure.text.serializer.legacy.LegacyComponentSerializer;
012import org.bukkit.entity.Player;
013import org.bukkit.event.inventory.ClickType;
014import org.bukkit.event.inventory.InventoryClickEvent;
015import org.bukkit.inventory.ItemStack;
016import org.spongepowered.configurate.ConfigurateException;
017import org.spongepowered.configurate.serialize.SerializationException;
018
019import java.util.ArrayList;
020import java.util.List;
021
022public class SlotViewItem implements GUIItem {
023    private final CalibrePlugin plugin;
024    private final PaperSlot slot;
025    private final String slotKey;
026    private final CalibreComponent<BukkitItem> component;
027    private boolean modification;
028    private boolean limited;
029
030    public SlotViewItem(CalibrePlugin plugin, PaperSlot slot, String slotKey, CalibreComponent<BukkitItem> component, boolean modification, boolean limited) {
031        this.plugin = plugin;
032        this.slot = slot;
033        this.slotKey = slotKey;
034        this.component = component;
035        this.modification = modification;
036        this.limited = limited;
037    }
038
039    public SlotViewItem(CalibrePlugin plugin, PaperSlot slot, String slotKey, boolean modification, boolean limited) {
040        this.plugin = plugin;
041        this.slot = slot;
042        this.slotKey = slotKey;
043        this.component = slot.get();
044        this.modification = modification;
045        this.limited = limited;
046    }
047
048    public SlotViewItem(CalibrePlugin plugin, CalibreComponent<BukkitItem> component, boolean modification, boolean limited) {
049        this.plugin = plugin;
050        slot = null;
051        slotKey = null;
052        this.component = component;
053        this.modification = modification;
054        this.limited = limited;
055    }
056
057    public CalibrePlugin plugin() { return plugin; }
058
059    public PaperSlot slot() { return slot; }
060    public String slotKey() { return slotKey; }
061    public CalibreComponent<?> component() { return component; }
062
063    public boolean modification() { return modification; }
064    public void modification(boolean modification) { this.modification = modification; }
065
066    public boolean limited() { return limited; }
067    public void limited(boolean limited) { this.limited = limited; }
068
069    @Override
070    public ItemStack createItem(GUIView view) {
071        CalibreComponent<BukkitItem> component = this.component;
072        if (component == null) {
073            if (slot == null)
074                return null;
075            else
076                component = slot.get();
077        }
078
079        CalibreComponent<BukkitItem> cursor = null;
080        try {
081            cursor = plugin.getComponent(view.getView().getCursor());
082        } catch (ConfigurateException e) {
083            e.printStackTrace();
084        }
085
086        String locale = view.getPlayer().getLocale();
087        ItemStack item = null;
088        if (component == null) {
089            item = slot.createViewItem(locale, slotKey, cursor);
090        } else {
091            try {
092                CalibreComponent<BukkitItem> copy = component.copy();
093                copy.slots().clear();
094                ComponentTree tree = new ComponentTree(copy).buildTree();
095                tree.complete(false);
096                tree.buildStats();
097                item = copy.create(locale).item();
098            } catch (SerializationException e) {
099                e.printStackTrace();
100            }
101        }
102
103        if (item != null && slot != null && slot.fieldModifiable())
104            BukkitUtils.modMeta(item, meta -> {
105                List<String> lore = meta.getLore();
106                if (lore == null)
107                    lore = new ArrayList<>();
108                lore.add(LegacyComponentSerializer.legacySection().serialize(plugin.gen(locale, "slot_view.field_modifiable")));
109                meta.setLore(lore);
110            });
111        return item;
112    }
113
114    @Override
115    public void onClick(GUIView view, InventoryClickEvent event) {
116        event.setCancelled(true);
117
118        if (slot == null)
119            return;
120        if (!modification || (limited && !slot.fieldModifiable()))
121            return;
122
123        Player player = view.getPlayer();
124        String locale = player.getLocale();
125        ItemStack rawCursor = view.getRawCursor();
126        if (BukkitUtils.empty(rawCursor) && slot.get() != null) {
127            if (event.getClick() == ClickType.RIGHT) {
128                // TODO component editing menu
129                return;
130            }
131
132            view.setRawCursor(slot.<CalibreComponent<BukkitItem>>get().buildTree().create(locale).item());
133            slot.set(null);
134        } else if (!BukkitUtils.empty(rawCursor)) {
135            CalibreComponent<BukkitItem> component = null;
136            try {
137                component = plugin.getComponent(rawCursor);
138            } catch (ConfigurateException e) {
139                e.printStackTrace();
140            }
141            if (component == null || !slot.isCompatible(component))
142                return;
143
144            if (slot.get() == null)
145                view.setRawCursor(rawCursor.subtract());
146            else if (rawCursor.getAmount() > 1)
147                return;
148            else
149                view.setRawCursor(slot.<CalibreComponent<BukkitItem>>get().buildTree().create(locale).item());
150            slot.set(component);
151        } else
152            return;
153
154        event.setCancelled(false);
155        if (view.getGUI() instanceof SlotViewGUI)
156            ((SlotViewGUI) view.getGUI()).notifyUpdate(view);
157    }
158}
