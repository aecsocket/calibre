001package me.aecsocket.calibre.system;
002
003import me.aecsocket.calibre.component.CalibreComponent;
004import me.aecsocket.calibre.component.PaperComponent;
005import me.aecsocket.calibre.world.ItemSlot;
006import me.aecsocket.calibre.world.ItemUser;
007import me.aecsocket.calibre.wrapper.BukkitItem;
008import me.aecsocket.calibre.wrapper.slot.BukkitSlot;
009import me.aecsocket.calibre.wrapper.slot.EntityEquipmentSlot;
010import me.aecsocket.calibre.wrapper.user.PlayerUser;
011import me.aecsocket.unifiedframework.loop.TickContext;
012import org.bukkit.entity.Player;
013import org.bukkit.event.Event;
014import org.bukkit.event.inventory.InventoryClickEvent;
015import org.bukkit.event.player.PlayerInteractEvent;
016import org.bukkit.inventory.EquipmentSlot;
017
018public final class BukkitItemEvents {
019    private BukkitItemEvents() {}
020
021    public static class BukkitEquipped extends ItemEvents.Equipped<BukkitItem> {
022        private final TickContext tickContext;
023
024        public BukkitEquipped(CalibreComponent<BukkitItem> component, ItemUser user, ItemSlot<BukkitItem> slot, TickContext tickContext) {
025            super(component, user, slot);
026            this.tickContext = tickContext;
027        }
028
029        public TickContext tickContext() { return tickContext; }
030
031        public static BukkitEquipped of(PaperComponent component, Player player, EquipmentSlot slot, TickContext tickContext) {
032            return new BukkitEquipped(component, PlayerUser.of(player), EntityEquipmentSlot.of(player, slot), tickContext);
033        }
034    }
035
036    public static class BukkitInteract extends ItemEvents.Interact<BukkitItem> {
037        private final PlayerInteractEvent event;
038
039        public BukkitInteract(PlayerInteractEvent event, CalibreComponent<BukkitItem> component, ItemUser user, ItemSlot<BukkitItem> slot, int type) {
040            super(component, user, slot, type);
041            this.event = event;
042        }
043
044        public PlayerInteractEvent event() { return event; }
045
046        @Override public boolean cancelled() { return event.useItemInHand() == Event.Result.DENY; }
047        @Override public void cancel() { event.setUseItemInHand(Event.Result.DENY); }
048
049        public static BukkitInteract of(PlayerInteractEvent event, PaperComponent component, EquipmentSlot slot) {
050            Player player = event.getPlayer();
051            int type;
052            switch (event.getAction()) {
053                case LEFT_CLICK_AIR:
054                case LEFT_CLICK_BLOCK:
055                    type = ItemEvents.Interact.LEFT;
056                    break;
057                case RIGHT_CLICK_AIR:
058                case RIGHT_CLICK_BLOCK:
059                    type = ItemEvents.Interact.RIGHT;
060                    break;
061                default:
062                    throw new IllegalArgumentException("Incompatible action type " + event.getAction());
063            }
064            return new BukkitInteract(
065                    event,
066                    component,
067                    PlayerUser.of(player),
068                    EntityEquipmentSlot.of(player, slot),
069                    type
070            );
071        }
072    }
073
074    public static class BukkitClick extends ItemEvents.Click<BukkitItem> {
075        private final InventoryClickEvent event;
076
077        public BukkitClick(InventoryClickEvent event, CalibreComponent<BukkitItem> component, ItemUser user, ItemSlot<BukkitItem> slot, ItemSlot<BukkitItem> cursor, boolean leftClick, boolean rightClick, boolean shiftClick) {
078            super(component, user, slot, cursor, leftClick, rightClick, shiftClick);
079            this.event = event;
080        }
081
082        public InventoryClickEvent event() { return event; }
083
084        @Override public boolean cancelled() { return event.isCancelled(); }
085        @Override public void cancel() { event.setCancelled(true); }
086
087        public static BukkitClick of(InventoryClickEvent event, PaperComponent component) {
088            return new BukkitClick(
089                    event,
090                    component,
091                    PlayerUser.of((Player) event.getWhoClicked()),
092                    BukkitSlot.of(event::getCurrentItem, event::setCurrentItem),
093                    BukkitSlot.of(event::getCursor, event.getView()::setCursor),
094                    event.isLeftClick(), event.isRightClick(), event.isShiftClick()
095            );
096        }
097    }
098}
