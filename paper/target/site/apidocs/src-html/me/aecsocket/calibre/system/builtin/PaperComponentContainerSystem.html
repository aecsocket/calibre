001package me.aecsocket.calibre.system.builtin;
002
003import me.aecsocket.calibre.CalibrePlugin;
004import me.aecsocket.calibre.wrapper.BukkitItem;
005import net.kyori.adventure.text.Component;
006import org.spongepowered.configurate.objectmapping.ConfigSerializable;
007
008@ConfigSerializable
009public class PaperComponentContainerSystem extends ComponentContainerSystem<BukkitItem> {
010    private transient final CalibrePlugin plugin;
011
012    public PaperComponentContainerSystem(CalibrePlugin plugin) {
013        this.plugin = plugin;
014    }
015
016    public PaperComponentContainerSystem() { this(CalibrePlugin.getInstance()); }
017
018    public PaperComponentContainerSystem(ComponentContainerSystem<BukkitItem> o, CalibrePlugin plugin) {
019        super(o);
020        this.plugin = plugin;
021    }
022
023    public PaperComponentContainerSystem(PaperComponentContainerSystem o) {
024        this(o, o.plugin);
025    }
026
027    public CalibrePlugin plugin() { return plugin; }
028    @Override public Component localize(String locale, String key, Object... args) { return plugin.gen(locale, key, args); }
029
030    @Override protected int listenerPriority() { return plugin.setting("system", ID, "component_container").getInt(1500); }
031
032    @Override public PaperComponentContainerSystem copy() { return new PaperComponentContainerSystem(this); }
033}
