001package me.aecsocket.calibre.system.builtin;
002
003import me.aecsocket.calibre.CalibrePlugin;
004import net.kyori.adventure.text.Component;
005import org.spongepowered.configurate.objectmapping.ConfigSerializable;
006
007@ConfigSerializable
008public class PaperNameFromChildSystem extends NameFromChildSystem {
009    private transient final CalibrePlugin plugin;
010
011    public PaperNameFromChildSystem(CalibrePlugin plugin) {
012        this.plugin = plugin;
013    }
014
015    public PaperNameFromChildSystem() { this(CalibrePlugin.getInstance()); }
016
017    public PaperNameFromChildSystem(NameFromChildSystem o, CalibrePlugin plugin) {
018        super(o);
019        this.plugin = plugin;
020    }
021
022    public PaperNameFromChildSystem(PaperNameFromChildSystem o) {
023        this(o, o.plugin);
024    }
025
026    public CalibrePlugin plugin() { return plugin; }
027    @Override public Component localize(String locale, String key, Object... args) { return plugin.gen(locale, key, args); }
028
029    @Override protected int listenerPriority() { return plugin.setting("system", ID, "listener_priority").getInt(1500); }
030
031    @Override public PaperNameFromChildSystem copy() { return new PaperNameFromChildSystem(this); }
032}
