001package me.aecsocket.calibre.system.builtin;
002
003import me.aecsocket.calibre.CalibrePlugin;
004import net.kyori.adventure.text.Component;
005import org.bukkit.map.MapFont;
006import org.spongepowered.configurate.objectmapping.ConfigSerializable;
007
008import java.util.HashMap;
009import java.util.Map;
010
011@ConfigSerializable
012public class PaperStatDisplaySystem extends StatDisplaySystem {
013    private transient final CalibrePlugin plugin;
014    private static final Map<String, String> padding = new HashMap<>();
015
016    public PaperStatDisplaySystem(CalibrePlugin plugin) {
017        this.plugin = plugin;
018    }
019
020    public PaperStatDisplaySystem() { this(CalibrePlugin.getInstance()); }
021
022    public PaperStatDisplaySystem(StatDisplaySystem o, CalibrePlugin plugin) {
023        super(o);
024        this.plugin = plugin;
025    }
026
027    public PaperStatDisplaySystem(PaperStatDisplaySystem o) {
028        this(o, o.plugin);
029    }
030
031    public CalibrePlugin plugin() { return plugin; }
032    @Override public Component localize(String locale, String key, Object... args) { return plugin.gen(locale, key, args); }
033
034    @Override
035    protected int listenerPriority() {
036        return plugin.setting("system", ID, "listener_priority").getInt(1000);
037    }
038
039    @Override protected int getWidth(String text) {
040        MapFont font = plugin.getFont();
041        if (!font.isValid(text))
042            throw new IllegalArgumentException("Invalid characters in text: not added to font map?");
043        return font.getWidth(text);
044    }
045
046    private String padding(String locale) {
047        return padding.computeIfAbsent(locale, __ -> plugin.setting("symbol", "pad").getString(" "));
048    }
049
050    @Override
051    protected String pad(String locale, int width) {
052        String padding = padding(locale);
053        MapFont font = plugin.getFont();
054        if (font.isValid(padding)) {
055            String pad = "";
056            String nextPad = padding;
057            while (font.getWidth(nextPad) < width) {
058                pad = nextPad;
059                nextPad += padding;
060            }
061            return pad;
062        }
063        return padding.repeat(width);
064    }
065
066    @Override public PaperStatDisplaySystem copy() { return new PaperStatDisplaySystem(this); }
067}
