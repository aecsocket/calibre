001package me.aecsocket.calibre.util;
002
003import com.comphenix.protocol.PacketType;
004import com.comphenix.protocol.events.PacketContainer;
005import me.aecsocket.calibre.CalibrePlugin;
006import org.bukkit.GameMode;
007import org.bukkit.entity.Player;
008
009public final class CalibreProtocol {
010    private CalibreProtocol() {}
011
012    private static CalibrePlugin plugin() { return CalibrePlugin.getInstance(); }
013
014    public static void fov(Player target, double fov) {
015        CalibrePlugin plugin = plugin();
016        PacketContainer packet = plugin.getProtocol().createPacket(PacketType.Play.Server.ABILITIES);
017        packet.getBooleans().write(0, target.isInvulnerable());
018        packet.getBooleans().write(1, target.isFlying());
019        packet.getBooleans().write(2, target.getAllowFlight());
020        packet.getBooleans().write(3, target.getGameMode() == GameMode.CREATIVE);
021        packet.getFloat().write(0, target.getFlySpeed() / 2);
022        packet.getFloat().write(1, (float) fov);
023        plugin.sendPacket(packet, target);
024    }
025}
