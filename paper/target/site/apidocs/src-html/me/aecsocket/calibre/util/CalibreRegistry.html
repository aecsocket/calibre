001package me.aecsocket.calibre.util;
002
003import me.aecsocket.unifiedframework.registry.Registry;
004
005import java.util.ArrayList;
006import java.util.List;
007import java.util.function.Consumer;
008
009public class CalibreRegistry extends Registry<CalibreIdentifiable> {
010    private final List<Consumer<CalibreRegistry>> onLoad = new ArrayList<>();
011
012    public List<Consumer<CalibreRegistry>> onLoad() { return onLoad; }
013    public void onLoad(Consumer<CalibreRegistry> supplier) { onLoad.add(supplier); }
014
015    @Override
016    public void unregisterAll() {
017        super.unregisterAll();
018        onLoad.forEach(consumer -> consumer.accept(this));
019    }
020}
