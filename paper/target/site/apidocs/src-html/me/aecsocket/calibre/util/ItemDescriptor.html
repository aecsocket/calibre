001package me.aecsocket.calibre.util;
002
003import me.aecsocket.unifiedframework.stat.AbstractStat;
004import me.aecsocket.unifiedframework.stat.serialization.ConfigurateStat;
005import me.aecsocket.unifiedframework.stat.serialization.FunctionCreationException;
006import me.aecsocket.unifiedframework.util.BukkitUtils;
007import org.bukkit.Material;
008import org.bukkit.NamespacedKey;
009import org.bukkit.Registry;
010import org.bukkit.inventory.ItemStack;
011import org.bukkit.inventory.meta.Damageable;
012import org.jetbrains.annotations.NotNull;
013import org.spongepowered.configurate.ConfigurationNode;
014import org.spongepowered.configurate.objectmapping.ConfigSerializable;
015import org.spongepowered.configurate.serialize.SerializationException;
016
017import java.lang.reflect.Type;
018import java.util.function.Function;
019
020@ConfigSerializable
021public class ItemDescriptor {
022    public static class Stat extends AbstractStat<ItemDescriptor> implements ConfigurateStat<ItemDescriptor> {
023        public Stat(ItemDescriptor defaultValue) { super(defaultValue); }
024        public Stat() {}
025        @Override public Type getValueType() { return ItemDescriptor.class; }
026        @Override public ItemDescriptor copy(@NotNull ItemDescriptor obj) { return obj; }
027
028        @Override
029        public Function<ItemDescriptor, ItemDescriptor> getModFunction(ConfigurationNode node) throws FunctionCreationException {
030            ItemDescriptor value;
031            try {
032                value = node.get(ItemDescriptor.class);
033            } catch (SerializationException e) {
034                throw new FunctionCreationException(e);
035            }
036            return b -> value;
037        }
038
039        @Override protected Stat createNew() { return new Stat(); }
040    }
041
042    private final NamespacedKey id;
043    private final Integer modelData;
044    private final Integer damage;
045
046    public ItemDescriptor(NamespacedKey id, Integer modelData, Integer damage) {
047        this.id = id;
048        this.modelData = modelData;
049        this.damage = damage;
050    }
051
052    public ItemDescriptor() { this(null, null, null); }
053
054    public NamespacedKey id() { return id; }
055    public ItemDescriptor id(NamespacedKey id) { return new ItemDescriptor(id, modelData, damage); }
056
057    public Integer modelData() { return modelData; }
058    public ItemDescriptor modelData(Integer modelData) { return new ItemDescriptor(id, modelData, damage); }
059
060    public Integer damage() { return damage; }
061    public ItemDescriptor damage(Integer damage) { return new ItemDescriptor(id, modelData, damage); }
062
063    public ItemStack apply(ItemStack item) {
064        Material material = Registry.MATERIAL.get(id);
065        if (material == null)
066            throw new IllegalArgumentException("Invalid item key " + id);
067        item.setType(material);
068        return BukkitUtils.modMeta(item, meta -> {
069            if (modelData != null)
070                meta.setCustomModelData(modelData);
071            if (damage != null && meta instanceof Damageable)
072                ((Damageable) meta).setDamage(damage);
073        });
074    }
075
076    public ItemStack create(int amount) {
077        return apply(new ItemStack(Material.AIR, amount));
078    }
079
080    public ItemStack create() { return create(1); }
081
082    @Override
083    public String toString() {
084        return "ItemDescriptor{" +
085                "id=" + id +
086                ", modelData=" + modelData +
087                ", damage=" + damage +
088                '}';
089    }
090}
