001package me.aecsocket.calibre.wrapper.slot;
002
003import me.aecsocket.calibre.world.ItemSlot;
004import me.aecsocket.calibre.wrapper.BukkitItem;
005import me.aecsocket.unifiedframework.util.BukkitUtils;
006import org.bukkit.inventory.ItemStack;
007
008import java.util.function.Consumer;
009import java.util.function.Supplier;
010
011public interface BukkitSlot extends ItemSlot<BukkitItem> {
012    ItemStack bukkitGet();
013    void bukkitSet(ItemStack item);
014
015    @Override default BukkitItem get() {
016        ItemStack stack = bukkitGet();
017        return BukkitUtils.empty(stack) ? null : BukkitItem.of(stack);
018    }
019    @Override default void set(BukkitItem item) { bukkitSet(item == null ? null : item.item()); }
020
021    static BukkitSlot of(Supplier<ItemStack> get, Consumer<ItemStack> set) {
022        return new BukkitSlot() {
023            @Override public ItemStack bukkitGet() { return get.get(); }
024            @Override public void bukkitSet(ItemStack item) { set.accept(item); }
025        };
026    }
027}
