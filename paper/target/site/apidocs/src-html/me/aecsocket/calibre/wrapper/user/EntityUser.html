001package me.aecsocket.calibre.wrapper.user;
002
003import me.aecsocket.calibre.CalibrePlugin;
004import me.aecsocket.calibre.world.ItemUser;
005import me.aecsocket.calibre.world.SenderUser;
006import me.aecsocket.unifiedframework.util.VectorUtils;
007import me.aecsocket.unifiedframework.util.vector.Vector3D;
008import net.kyori.adventure.text.Component;
009import org.bukkit.entity.Entity;
010
011public interface EntityUser extends ItemUser, SenderUser {
012    Entity entity();
013
014    @Override
015    default void sendMessage(Component component) {
016        CalibrePlugin.getInstance().getAudiences()
017                .sender(entity())
018                .sendMessage(component);
019    }
020
021    @Override
022    default void sendInfo(Component component) {
023        CalibrePlugin.getInstance().getAudiences()
024                .sender(entity())
025                .sendActionBar(component);
026    }
027
028    @Override default Vector3D position() { return VectorUtils.toUF(entity().getLocation().toVector()); }
029    @Override default Vector3D direction() { return VectorUtils.toUF(entity().getLocation().getDirection()); }
030}
